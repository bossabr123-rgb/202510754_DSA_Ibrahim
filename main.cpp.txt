#include "student.h"

int main () {

    int choice ; 

    do {

        cout << "\n========== STUDENT MANAGEMENT SYSTEM ==========\n" ;
        cout << "1. Add Student" << endl ; 
        cout << "2. Display Students" << endl ; 
        cout << "3. Search Student" << endl ; 
        cout << "4. Update Student" << endl ; 
        cout << "5. Delete Student" << endl ; 
        cout << "6. Exit" << endl ; 

        cout << "\nEnter your Choice : " ;
        cin >> choice ; 

        switch (choice) {

            case 1 :
                addStudent () ; 
                break ; 

            case 2 :
                displayStudents () ; 
                break ; 

            case 3 :
                searchStudent () ; 
                break ; 

            case 4 :
                updateStudent () ; 
                break ; 

            case 5 :
                deleteStudent () ; 
                break ; 

            case 6 :
                cout << "\033[32m" ;
                cout << "Program Ended Successfully\n" ;
                cout << "\033[0m" ;
                break ; 

            default :
                cout << "\033[31m" ;
                cout << "Invalid Choice\n" ;
                cout << "\033[0m" ;
        }

    } while (choice != 6 ) ;

    return 0 ; 
}
